\chapter{Underlying transformations}
\label{ch:underlying}

In this chapter, for any two naturals $n$ and $m$ with $m < n$, we provide a way to transform
single-sorted $n$-categories (or $\omega$-categories) into single-sorted $m$-categories by
\emph{forgetting} morphisms that are not $m$-cells in the original category.

Similarly, we provide a way to transform single-sorted $n$-functors (or $\omega$-functors) into
single-sorted $m$-functors by forgetting the action on morphisms that are not $m$-cells in the
original category.

We conclude the chapter by showing that these transformations are functorial, i.e., they define a
functor from $n\Cat$ (or $\omega\Cat$) to $m\Cat$.

\section{Underlying single-sorted categories}

\begin{lemma}
  \label{lem:underlying_source_is_cell}
  \uses{def:cell}
  \lean{HigherCategoryTheory.underlying_source_is_cell}
  \leanok
  Let $\cat{C}$ be a single-sorted $\Index$-category and $m \in \Index$. Then, for every $f \in
  \cat{C}$ and every $k \in \Index$ such that $k < m$, we have that $\Sc^k (f)$ is a $m$-cell in
  $\cat{C}$.
\end{lemma}

\begin{proof}
  \leanok
  Applying one of the axioms of single-sorted categories, we have that
  \[
    \Sc^m (\Sc^k f) = \Sc^k (f).
  \]
\end{proof}

\begin{lemma}
  \label{lem:underlying_target_is_cell}
  \uses{thm:cell_sc_iff_cell_tg}
  \lean{HigherCategoryTheory.underlying_target_is_cell}
  \leanok
  Let $\cat{C}$ be a single-sorted $\Index$-category and $m \in \Index$. Then, for every $f \in
  \cat{C}$ and every $k \in \Index$ such that $k < m$, we have that $\Tg^k (f)$ is a $m$-cell in
  $\cat{C}$.
\end{lemma}

\begin{proof}
  \leanok
  Applying one of the axioms of single-sorted categories, we have that
  \[
    \Tg^m (\Tg^k f) = \Tg^k (f).
  \]
  That is, $\Tg^k (f)$ is a target-based $m$-cell in $\cat{C}$. However, we know that every
  target-based $m$-cell is also a $m$-cell.
\end{proof}

\begin{lemma}
  \label{lem:underlying_comp_is_cell}
  \uses{def:cell}
  \lean{HigherCategoryTheory.underlying_comp_is_cell}
  \leanok
  Let $\cat{C}$ be a single-sorted $\Index$-category and $m \in \Index$. Then, for every $f, g \in
  C_m$ (i.e., $f$ and $g$ are $m$-cells in $\cat{C}$) and every $k \in \Index$ such that $k < m$,
  if the comoposition $g \Pcomp^k f$ is defined, then $g \Pcomp^k f$ is a $m$-cell in $\cat{C}$.
\end{lemma}

\begin{proof}
  \leanok
  Applying one of the axioms of single-sorted categories and the fact that $f$ and $g$ are
  $m$-cells, we have that
  \[
    \Sc^m (g \Pcomp^k f) = \Sc^m (g) \Pcomp^k \Sc^m (f) = g \Pcomp^k f.
  \]
\end{proof}

\begin{definition}[Underlying single-sorted category]
  \label{def:SingleSortedCategory_underlying}
  \uses{
    lem:underlying_source_is_cell,
    lem:underlying_target_is_cell,
    lem:underlying_comp_is_cell,
  }
  \lean{
    HigherCategoryTheory.SingleSortedNCategory.underlying,
    HigherCategoryTheory.SingleSortedOmegaCategory.underlying,
  }
  \leanok
  Let $\cat{C}$ be a single-sorted $n$-category (or $\omega$-category) and $m$ be a natural
  number such that $m < n$\footnote{If $\cat{C}$ is an $\omega$-category, then $m$ can be any
  natural number.}. We define the \emph{underlying single-sorted $m$-category} of $\cat{C}$ as
  the single-sorted $m$-category $\cat{C}_m$ defined on the set of $m$-cells of $\cat{C}$, $C_m$,
  with the corestriction of the source, target, identity, and composition operations of $\cat{C}$ to
  $C_m$.

  Note that, by Lemmas~\ref{lem:underlying_source_is_cell}, \ref{lem:underlying_target_is_cell}, and
  \ref{lem:underlying_comp_is_cell}, these operations are well-defined. Also, the axioms of
  single-sorted categories for $\cat{C}_m$ follow directly from the corresponding axioms for
  $\cat{C}$.
\end{definition}

\section{Underlying single-sorted functors}

\begin{lemma}
  \label{lem:underlying_functor_is_cell}
  \uses{def:cell, def:SingleSortedFunctor}
  \lean{HigherCategoryTheory.underlying_functor_is_cell}
  \leanok
  Let $\cat{C}$ and $\cat{D}$ be single-sorted $\Index$-categories, $F \colon \cat{C} \to
  \cat{D}$ be a single-sorted functor, and $m \in \Index$. Then, for every $f \in C_m$ we have that
  $F(f) \in D_m$, i.e., the image of $m$-cells under a single-sorted functor are $m$-cells.
\end{lemma}

\begin{proof}
  \leanok
  Applying one of the axioms of single-sorted functors and the fact that $f$ is a $m$-cell, we
  have that
  \[
    \Sc^m (F(f)) = F(\Sc^m (f)) = F(f).
  \]
\end{proof}

\begin{definition}[Underlying single-sorted functor]
  \label{def:SingleSortedFunctor_underlying}
  \uses{lem:underlying_functor_is_cell}
  \lean{
    HigherCategoryTheory.SingleSortedNFunctor.underlying,
    HigherCategoryTheory.SingleSortedOmegaFunctor.underlying,
  }
  \leanok
  Let $\cat{C}$ and $\cat{D}$ be single-sorted $\Index$-categories, $F \colon \cat{C} \to
  \cat{D}$ be a single-sorted functor, and $m \in \Index$. We define the
  \emph{underlying single-sorted $m$-functor} of $F$ as the single-sorted $m$-functor $F_m \colon
  \cat{C}_m \to \cat{D}_m$ defined as the corestriction of $F$ to the sets of $m$-cells
  $C_m$ and $D_m$.

  Note that, by Lemma~\ref{lem:underlying_functor_is_cell}, this function is well-defined. Also, the
  axioms of single-sorted functors for $F_m$ follow directly from the corresponding axioms for $F$.
\end{definition}
